const isToday = (date: string | Date) => {
  const today = new Date();
  const inputDate = new Date(date);
  return (
    today.getFullYear() === inputDate.getFullYear() &&
    today.getMonth() === inputDate.getMonth() &&
    today.getDate() === inputDate.getDate()
  );
};
